colorise.js

A small JavaScript library for color manipulation.

Installation using npm

npm i colorise --save


Usage

const colorise = require('colorise');

// create instance of class library
let c = new colorise();

// get color-name from hex-color-code
let result = c.getColorName('#714693');
console.log(result); // 'Affair'

// get hex-color-code from color-name
result = c.getColorCode('Black Rose');
console.log(result); // '#67032d'

// here are different conversions returns array or object
c.rgbToHex(50,168,82); // returns hex-code
c.hexToRgb(#165852); // returns r,g,b values
c.rgbToHsl(110,154,111); // returns new set of h,s,l
c.negateColor(111,120,145); // returns negated color
